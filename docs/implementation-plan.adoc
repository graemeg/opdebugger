== Introduction

=== Purpose

This document provides a detailed implementation plan for the Object Pascal Debug Format (OPDF) and Pascal Debug Reference (PDR) debugger. The plan follows an iterative "vertical slice" approach where each iteration delivers end-to-end functionality for a specific feature set.

=== Strategy

**Concurrent Development**: Implement OPDF compiler support and PDR debugger simultaneously, validating each feature as it is added.

**Test-Driven Development**: Write tests first to prove functionality works before implementation.

**Vertical Slices**: Each phase delivers working functionality for a specific type or feature, from compiler generation through debugger display.

unit ogopdf;

{$mode objfpc}{$H+}

interface

uses
  SysUtils;

const
  OPDF_MAGIC = 'OPDF';
  OPDF_VERSION = 1;

type
  { OPDF File Header }
  TOPDFHeader = packed record
    Magic: array[0..3] of Char;  // 'OPDF'
    Version: Word;               // Format version
    BuildID: TGUID;              // UUID for binary linking
    TotalRecords: Cardinal;      // Number of records
    Flags: Cardinal;             // Feature flags
  end;

  { Record Types }
  TOPDFRecordType = (
    recUnknown       = 0,
    recPrimitive     = 1,
    recGlobalVar     = 2,
    recShortStr      = 3,
    recAnsiStr       = 4,
    recUnicodeStr    = 5,
    recPointer       = 6,
    recArray         = 7,
    recRecord        = 8,
    recClass         = 9,
    recProperty      = 10,
    recMethod        = 11,
    recLocalVar      = 12,
    recParameter     = 13
  );

  { Generic Record Header }
  TOPDFRecordHeader = packed record
    RecType: Byte;
    RecSize: Word;
  end;

  { Primitive Type Definition }
  TDefPrimitive = packed record
    TypeID: Word;
    SizeInBytes: Byte;
    IsSigned: Byte;      // 1=True, 0=False
    NameLen: Byte;
    // Followed by Name string
  end;

  { Global Variable Symbol }
  TDefGlobalVar = packed record
    TypeID: Word;
    Address: QWord;
    NameLen: Byte;
    // Followed by Name string
  end;

  { ShortString Type }
  TDefShortString = packed record
    TypeID: Word;
    MaxLength: Byte;
    NameLen: Byte;
    // Followed by Name string
  end;

  { AnsiString Type }
  TDefAnsiString = packed record
    TypeID: Word;
    NameLen: Byte;
    // Followed by Name string
  end;

  { Pointer Type }
  TDefPointer = packed record
    TypeID: Word;
    TargetTypeID: Word;
    NameLen: Byte;
    // Followed by Name string
  end;

  { Class Definition }
  TDefClass = packed record
    TypeID: Word;
    ParentTypeID: Word;
    VMTAddress: QWord;
    InstanceSize: Cardinal;
    FieldCount: Word;
    NameLen: Byte;
    // Followed by Name string
    // Followed by field descriptors
  end;

  { Property Definition }
  TDefProperty = packed record
    ClassTypeID: Word;
    PropertyTypeID: Word;
    ReadType: Byte;      // 0=Field, 1=Method
    WriteType: Byte;     // 0=Field, 1=Method, 2=None
    ReadAddr: QWord;
    WriteAddr: QWord;
    NameLen: Byte;
    // Followed by Name string
  end;

implementation

end.

    ESI, EDI, EBP, ESP: Cardinal;
    EIP: Cardinal;
    {$ENDIF}
  end;

  IProcessController = interface
    ['{A1B2C3D4-E5F6-7890-ABCD-EF1234567890}']
    function Attach(PID: Integer): Boolean;
    function Detach: Boolean;
    function Continue: Boolean;
    function Step: Boolean;

                       out Buffer): Boolean;
    function WriteMemory(Address: QWord; Size: Cardinal;

  end;

  Payload: TDefPrimitive;
begin
  Payload.TypeID := TypeID;
  Payload.SizeInBytes := Size;
  Payload.IsSigned := Ord(Signed);
  Payload.NameLen := Length(Name);

  Header.RecSize := SizeOf(TDefPrimitive) + Length(Name);

  FStream.Write(Header, SizeOf(Header));
  FStream.Write(Payload, SizeOf(Payload));
  if Length(Name) > 0 then

  Inc(FRecordCount);
end;

unit pdr_loader;

{$mode objfpc}{$H+}

interface

uses
  Classes, SysUtils, ogopdf;

type
  TOPDFSection = class
  private
    FStream: TMemoryStream;
    FHeader: TOPDFHeader;
  public
    function LoadFromFile(const FileName: String): Boolean;
    function LoadFromELFSection(const BinaryPath: String): Boolean;

    property Header: TOPDFHeader read FHeader;
    property Stream: TMemoryStream read FStream;
  end;

implementation

function TOPDFSection.LoadFromFile(const FileName: String): Boolean;
var
  FS: TFileStream;
begin
  Result := False;
  if not FileExists(FileName) then Exit;

  FS := TFileStream.Create(FileName, fmOpenRead);
  try
    FStream := TMemoryStream.Create;
    FStream.CopyFrom(FS, FS.Size);
    FStream.Position := 0;

    FStream.Read(FHeader, SizeOf(FHeader));
    Result := (FHeader.Magic = OPDF_MAGIC) and
              (FHeader.Version = OPDF_VERSION);
  finally
    FS.Free;
  end;
end;

end.

**File**: `pdr/core/pdr_parser.pas`

[source,pascal]

unit pdr_parser;

{$mode objfpc}{$H+}

interface

uses
  Classes, SysUtils, ogopdf, Generics.Collections;

type
  TTypeInfo = record
    TypeID: Word;
    Name: String;
    Size: Byte;
    IsSigned: Boolean;
  end;

  TVariableInfo = record
    Name: String;
    TypeID: Word;
    Address: QWord;
  end;

  TOPDFParser = class
  private
    FTypes: specialize TDictionary<Word, TTypeInfo>;
    FVariables: specialize TDictionary<String, TVariableInfo>;
  public
    constructor Create;
    destructor Destroy; override;

    procedure Parse(Stream: TStream);

    function FindVariable(const Name: String;
                         out VarInfo: TVariableInfo): Boolean;
    function FindType(TypeID: Word;
                     out TypeInfo: TTypeInfo): Boolean;
  end;

implementation

procedure TOPDFParser.Parse(Stream: TStream);
var
  RecHeader: TOPDFRecordHeader;
  Primitive: TDefPrimitive;
  GlobalVar: TDefGlobalVar;
  TypeInfo: TTypeInfo;
  VarInfo: TVariableInfo;
  Name: String;
begin
  while Stream.Position < Stream.Size do
  begin
    Stream.Read(RecHeader, SizeOf(RecHeader));

    case TOPDFRecordType(RecHeader.RecType) of
      recPrimitive:
      begin
        Stream.Read(Primitive, SizeOf(Primitive));
        SetLength(Name, Primitive.NameLen);
        Stream.Read(Name[1], Primitive.NameLen);

        TypeInfo.TypeID := Primitive.TypeID;
        TypeInfo.Name := Name;
        TypeInfo.Size := Primitive.SizeInBytes;
        TypeInfo.IsSigned := (Primitive.IsSigned = 1);
        FTypes.Add(TypeInfo.TypeID, TypeInfo);
      end;

      recGlobalVar:
      begin
        Stream.Read(GlobalVar, SizeOf(GlobalVar));
        SetLength(Name, GlobalVar.NameLen);
        Stream.Read(Name[1], GlobalVar.NameLen);

        VarInfo.Name := Name;
        VarInfo.TypeID := GlobalVar.TypeID;
        VarInfo.Address := GlobalVar.Address;
        FVariables.Add(Name, VarInfo);
      end;
    end;
  end;
end;

end.
----

unit pdr_ptrace_linux;

{$mode objfpc}{$H+}

interface

uses
  BaseUnix, pdr_control, SysUtils;

type
  TLinuxPtraceController = class(TInterfacedObject, IProcessController)
  private
    FPID: Integer;
    FAttached: Boolean;
  public
    function Attach(PID: Integer): Boolean;
    function Detach: Boolean;
    function Continue: Boolean;
    function Step: Boolean;
    function ReadMemory(Address: QWord; Size: Cardinal;
                       out Buffer): Boolean;
    function WriteMemory(Address: QWord; Size: Cardinal;
                        const Buffer): Boolean;
    function GetRegisters(out Regs: TRegisters): Boolean;
    function SetBreakpoint(Address: QWord): Boolean;
  end;

implementation

const
  PTRACE_ATTACH = 16;
  PTRACE_DETACH = 17;
  PTRACE_PEEKDATA = 2;
  PTRACE_CONT = 7;

function ptrace(request: cInt; pid: TPid; addr: Pointer;
               data: Pointer): cLong; cdecl; external 'c' name 'ptrace';

function TLinuxPtraceController.Attach(PID: Integer): Boolean;
begin
  Result := False;
  FPID := PID;
  if ptrace(PTRACE_ATTACH, FPID, nil, nil) = 0 then
  begin
    FAttached := True;
    waitpid(FPID, nil, 0); // Wait for stop
    Result := True;
  end;
end;

function TLinuxPtraceController.ReadMemory(Address: QWord;
                                          Size: Cardinal;
                                          out Buffer): Boolean;
var
  Data: cLong;
  P: PByte;
  I: Cardinal;
begin
  Result := False;
  if not FAttached then Exit;

  P := @Buffer;
  for I := 0 to (Size div SizeOf(cLong)) - 1 do
  begin
    Data := ptrace(PTRACE_PEEKDATA, FPID,
                  Pointer(Address + I * SizeOf(cLong)), nil);
    Move(Data, P^, SizeOf(cLong));
    Inc(P, SizeOf(cLong));
  end;

  Result := True;
end;

end.
----

=== 1.5: PDR CLI Interface

**File**: `pdr/cli/pdr_main.pas`

[source,pascal]
----
program pdr;

{$mode objfpc}{$H+}

uses
  SysUtils, Classes, pdr_loader, pdr_parser, pdr_control;

var
  Loader: TOPDFSection;
  Parser: TOPDFParser;
  Ctrl: IProcessController;
  BinaryPath: String;

procedure PrintVariable(const VarName: String);
var
  VarInfo: TVariableInfo;
  TypeInfo: TTypeInfo;
  Buffer: array[0..7] of Byte;
  IntVal: LongInt;
begin
  if not Parser.FindVariable(VarName, VarInfo) then
  begin
    WriteLn('Variable not found: ', VarName);
    Exit;
  end;

  if not Parser.FindType(VarInfo.TypeID, TypeInfo) then
  begin
    WriteLn('Type not found for variable');
    Exit;
  end;

  if not Ctrl.ReadMemory(VarInfo.Address, TypeInfo.Size, Buffer) then
  begin
    WriteLn('Failed to read memory');
    Exit;
  end;

  // For MVP: only handle 4-byte signed integer
  if (TypeInfo.Size = 4) and TypeInfo.IsSigned then
  begin
    Move(Buffer, IntVal, 4);
    WriteLn(VarName, ' = ', IntVal);
  end
  else
    WriteLn('Unsupported type');
end;

begin
  if ParamCount < 1 then
  begin
    WriteLn('Usage: pdr <binary>');
    Halt(1);
  end;

  BinaryPath := ParamStr(1);

  Loader := TOPDFSection.Create;
  Parser := TOPDFParser.Create;
  try
    // Load OPDF
    if not Loader.LoadFromFile(ChangeFileExt(BinaryPath, '.opdf')) then
    begin
      WriteLn('Failed to load debug info');
      Halt(1);
    end;

    Parser.Parse(Loader.Stream);

    WriteLn('PDR Debugger - OPDF MVP');
    WriteLn('Commands: print <var>, quit');
    WriteLn;

    // For MVP: assume process already running, get PID
    Write('Enter target PID: ');
    ReadLn(TargetPID);

    Ctrl := CreateProcessController;
    if not Ctrl.Attach(TargetPID) then
    begin
      WriteLn('Failed to attach to process');
      Halt(1);
    end;

    // Simple REPL
    while True do
    begin
      Write('(pdr) ');
      ReadLn(Command);

      if Command = 'quit' then Break;

      if Pos('print ', Command) = 1 then
      begin
        VarName := Copy(Command, 7, Length(Command));
        PrintVariable(Trim(VarName));
      end;
    end;

    Ctrl.Detach;
  finally
    Parser.Free;
    Loader.Free;
  end;
end.

=== 2.2: PDR String Display

**File**: `pdr/core/pdr_typesys.pas`

[source,pascal]
----
unit pdr_typesys;

{$mode objfpc}{$H+}

interface

uses
  pdr_control, pdr_parser;

type
  TDebugTypeSystem = class
  private
    FParser: TOPDFParser;
    FController: IProcessController;
  public
    constructor Create(AParser: TOPDFParser;
                      AController: IProcessController);

    function EvaluateVariable(const VarName: String): String;
  end;

implementation

function TDebugTypeSystem.EvaluateVariable(const VarName: String): String;
var
  VarInfo: TVariableInfo;
  TypeInfo: TTypeInfo;
  Buffer: array[0..255] of Byte;
  Len: Byte;
  Str: String;
  PtrVal: PAnsiChar;
  StrLen: LongInt;
begin
  Result := '<error>';

  if not FParser.FindVariable(VarName, VarInfo) then
    Exit('<not found>');

  if not FParser.FindType(VarInfo.TypeID, TypeInfo) then
    Exit('<type error>');

  // Handle ShortString
  if TypeInfo.Name = 'ShortString' then
  begin
    // Read length byte
    if not FController.ReadMemory(VarInfo.Address, 1, Len) then
      Exit('<read error>');

    // Read string data
    if not FController.ReadMemory(VarInfo.Address + 1, Len, Buffer) then
      Exit('<read error>');

    SetLength(Str, Len);
    Move(Buffer, Str[1], Len);
    Result := '''' + Str + '''';
  end

  // Handle AnsiString
  else if TypeInfo.Name = 'AnsiString' then
  begin
    // Read pointer
    if not FController.ReadMemory(VarInfo.Address, SizeOf(Pointer),
                                 PtrVal) then
      Exit('<read error>');

    if PtrVal = nil then
      Exit('''''');

    // Read length from heap (at Ptr - 8)
    if not FController.ReadMemory(QWord(PtrVal) - 8, 4, StrLen) then
      Exit('<read error>');

    // Read string data
    if StrLen > 255 then StrLen := 255; // Safety
    if not FController.ReadMemory(QWord(PtrVal), StrLen, Buffer) then
      Exit('<read error>');

    SetLength(Str, StrLen);
    Move(Buffer, Str[1], StrLen);
    Result := '''' + Str + '''';
  end;
end;

end.
----

== Conclusion

This implementation plan provides a concrete, phased approach to developing the Object Pascal Debug Format (OPDF) and Pascal Debug Reference (PDR) debugger. By following the vertical slice strategy, each phase delivers tangible, testable value while building toward the ultimate goal: a debugging experience for Free Pascal that rivals commercial alternatives.

The key to success is concurrent development of the format and debugger, ensuring the specification remains practical and grounded in real implementation requirements. The phased approach allows for early validation and course correction while managing risk and resource investment.

Upon completion, the Free Pascal community will have a native, first-class debugging solution optimized for Object Pascal's unique language features, finally achieving parity with Delphi's industry-leading debugging experience.
